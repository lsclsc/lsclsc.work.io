#### mac终端命令

 ```bash
  cd 目录名称 # 进入指定的目录
  cd .. #返回上一级目录
  cd ~ #进入当前用户的根目录
  pwd #查看当前所在的目录
  mkdir 目录名称 # 创建一个目录
  cp 原文件名字 目标文件名字 #赋值文件
  mv 原文件名字 目标名字 #移动文件
  rm 文件名字 #删除文件
  vim/vi 文件名 #通过vim/vi编辑器打开文件   
 ```
